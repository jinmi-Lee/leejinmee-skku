# app.py
# Korean Variety Show Trend Dashboard
# Run:
# streamlit run app.py

import streamlit as st
import pandas as pd
import plotly.express as px
import plotly.graph_objects as go
from datetime import datetime

# ----------------------------------
# PAGE CONFIG
# ----------------------------------
st.set_page_config(
    page_title="Korean Variety Show Trend Dashboard",
    page_icon="📺",
    layout="wide"
)

# ----------------------------------
# DARK THEME STYLING
# ----------------------------------
st.markdown("""
<style>
.main {
    background-color: #0f172a;
}

.stApp {
    background: linear-gradient(135deg,#0f172a,#111827);
}

h1, h2, h3 {
    color: white;
}

.metric-card {
    background-color:#1e293b;
    padding:15px;
    border-radius:15px;
}

[data-testid="stSidebar"] {
    background-color:#111827;
}

.block-container {
    padding-top: 2rem;
}
</style>
""", unsafe_allow_html=True)

# ----------------------------------
# HEADER
# ----------------------------------
st.title("📺 Korean Variety Show Trend Dashboard")
st.markdown("""
Analyze Korean variety show popularity, audience engagement,
cast influence, and online trends using interactive visualizations.
""")

# ----------------------------------
# SAMPLE DATA
# ----------------------------------
trend_data = pd.DataFrame({
    "Date": pd.date_range("2025-01-01", periods=12, freq="M"),
    "Running Man":[75,78,82,80,83,87,89,88,90,92,91,95],
    "Knowing Bros":[60,62,63,65,66,68,67,70,72,73,74,76],
    "2 Days 1 Night":[68,70,69,72,75,77,76,79,80,81,83,84]
})

cast_data = pd.DataFrame({
    "Cast Member":[
        "Yoo Jae-suk",
        "Kim Jong-kook",
        "Haha",
        "Lee Soo-geun",
        "Kang Ho-dong",
        "DinDin"
    ],
    "Popularity Score":[98,92,81,85,88,75]
})

keyword_data = pd.DataFrame({
    "Keyword":[
        "Funny Edit",
        "Legendary Moment",
        "Mission Challenge",
        "Cast Chemistry",
        "Meme Clip",
        "Reaction Scene"
    ],
    "Mentions":[3500,4200,2900,5100,4600,3100]
})

engagement_data = pd.DataFrame({
    "Show":[
        "Running Man",
        "Knowing Bros",
        "2 Days 1 Night"
    ],
    "Views":[5200000,4200000,3800000],
    "Likes":[410000,290000,240000],
    "Comments":[32000,26000,21000]
})

comparison_data = pd.DataFrame({
    "Show":["Running Man","Knowing Bros","2 Days 1 Night"],
    "YouTube Score":[95,82,78],
    "Social Media Score":[90,76,72],
    "TV Rating Score":[75,81,88]
})

# ----------------------------------
# SIDEBAR
# ----------------------------------
st.sidebar.title("🎛 Dashboard Filters")

selected_show = st.sidebar.selectbox(
    "Select Show",
    ["All","Running Man","Knowing Bros","2 Days 1 Night"]
)

selected_platform = st.sidebar.selectbox(
    "Platform",
    ["All","YouTube","Google Trends","Social Media","TV Ratings"]
)

selected_cast = st.sidebar.selectbox(
    "Cast Member",
    ["All"] + cast_data["Cast Member"].tolist()
)

date_range = st.sidebar.date_input(
    "Date Range",
    value=(
        trend_data["Date"].min(),
        trend_data["Date"].max()
    )
)

st.sidebar.markdown("---")
st.sidebar.info("""
This dashboard uses sample data.

Future version:
- YouTube API
- Google Trends
- Reddit/Twitter Analysis
- Real-time updates
""")

# ----------------------------------
# PROJECT OVERVIEW
# ----------------------------------
st.header("📌 Project Overview")

st.markdown("""
### Korean Variety Show Trend Dashboard

This project explores how Korean variety shows become popular and spread
through YouTube, social media, and online communities.

The dashboard helps users:

- Track popularity trends
- Compare audience engagement
- Analyze cast member influence
- Discover trending keywords and memes
- Compare programs across platforms

Built with:
**Python, Streamlit, Pandas, Plotly**
""")

# ----------------------------------
# KPIs
# ----------------------------------
col1,col2,col3,col4 = st.columns(4)

col1.metric("Shows Tracked", "3")
col2.metric("Cast Members", "6")
col3.metric("Keywords", "6")
col4.metric("Platforms", "4")

st.divider()

# ----------------------------------
# POPULARITY TREND ANALYSIS
# ----------------------------------
st.header("📈 Popularity Trend Analysis")

trend_fig = px.line(
    trend_data,
    x="Date",
    y=["Running Man","Knowing Bros","2 Days 1 Night"],
    markers=True,
    template="plotly_dark",
    title="Popularity Trend Over Time"
)

st.plotly_chart(trend_fig, use_container_width=True)

# ----------------------------------
# CAST MEMBER RANKING
# ----------------------------------
st.header("🏆 Cast Member Ranking")

cast_fig = px.bar(
    cast_data.sort_values(
        "Popularity Score",
        ascending=True
    ),
    x="Popularity Score",
    y="Cast Member",
    orientation="h",
    template="plotly_dark",
    color="Popularity Score",
    title="Most Influential Cast Members"
)

st.plotly_chart(cast_fig, use_container_width=True)

# ----------------------------------
# KEYWORD & MEME ANALYSIS
# ----------------------------------
st.header("🔥 Keyword & Meme Analysis")

keyword_fig = px.treemap(
    keyword_data,
    path=["Keyword"],
    values="Mentions",
    color="Mentions",
    template="plotly_dark"
)

st.plotly_chart(keyword_fig, use_container_width=True)

# ----------------------------------
# VIEWER ENGAGEMENT ANALYSIS
# ----------------------------------
st.header("💬 Viewer Engagement Analysis")

engagement_fig = go.Figure()

engagement_fig.add_trace(
    go.Bar(
        name="Views",
        x=engagement_data["Show"],
        y=engagement_data["Views"]
    )
)

engagement_fig.add_trace(
    go.Bar(
        name="Likes",
        x=engagement_data["Show"],
        y=engagement_data["Likes"]
    )
)

engagement_fig.add_trace(
    go.Bar(
        name="Comments",
        x=engagement_data["Show"],
        y=engagement_data["Comments"]
    )
)

engagement_fig.update_layout(
    template="plotly_dark",
    barmode="group",
    title="Audience Engagement Metrics"
)

st.plotly_chart(
    engagement_fig,
    use_container_width=True
)

# ----------------------------------
# INTERACTIVE SHOW COMPARISON
# ----------------------------------
st.header("⚔️ Interactive Show Comparison")

selected_shows = st.multiselect(
    "Select Shows",
    comparison_data["Show"],
    default=comparison_data["Show"].tolist()
)

filtered_comparison = comparison_data[
    comparison_data["Show"].isin(selected_shows)
]

comparison_fig = go.Figure()

for _, row in filtered_comparison.iterrows():
    comparison_fig.add_trace(
        go.Scatterpolar(
            r=[
                row["YouTube Score"],
                row["Social Media Score"],
                row["TV Rating Score"]
            ],
            theta=[
                "YouTube",
                "Social Media",
                "TV Ratings"
            ],
            fill="toself",
            name=row["Show"]
        )
    )

comparison_fig.update_layout(
    template="plotly_dark",
    polar=dict(radialaxis=dict(visible=True)),
    title="Show Performance Comparison"
)

st.plotly_chart(
    comparison_fig,
    use_container_width=True
)

# ----------------------------------
# DATA TABLE
# ----------------------------------
st.header("📊 Sample Dataset")

tab1, tab2, tab3 = st.tabs([
    "Popularity",
    "Cast",
    "Engagement"
])

with tab1:
    st.dataframe(trend_data, use_container_width=True)

with tab2:
    st.dataframe(cast_data, use_container_width=True)

with tab3:
    st.dataframe(engagement_data, use_container_width=True)

# ----------------------------------
# FOOTER
# ----------------------------------
st.divider()

st.markdown("""
